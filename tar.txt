using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApplication2
{
    class Program2
    {
        static void Main(string[] args)
        {
            int n = int.Parse(Console.ReadLine());
            int[] triangulo = new int[3];
            for (int i = 0; i  n; i++)
            {
                for (int o = 0; o  3; o++)
                {
                    triangulo[o] = int.Parse(Console.ReadLine());
                }
                if (triangulo[0] == triangulo[1] && triangulo[0] == triangulo[2])
                    Console.WriteLine(É Um Triangulo Equilátero);
                else if (triangulo[0] == triangulo[1]  triangulo[1] == triangulo[2])
                    Console.WriteLine(É Um Triangulo Isósceles);
                else
                {
                    Console.WriteLine(É Um Triangulo Escaleno, compadre);
                    
                }
            }



        }
    }
}
